\begin{frame}
  \frametitle{Programming Languages}
  \begin{itemize}
    \item Languages are tools
    \item Can be specialized
      \begin{itemize}
        \item Hammer
        \item Screwdriver
        \item Saw
      \end{itemize}
    \item Can be general usage
      \begin{itemize}
        \item Swiss army knife
      \end{itemize}
    \item Use the right tool for the right job
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Goal: Correctness}
  \structure{Ada}
  \begin{itemize}
    \item Developed by military
    \item Used in avionics, air traffic control, railways, banking, military and space technology
  \end{itemize}
  \vskip2mm
  \structure{Coq}
  \begin{itemize}
    \item Allows to prove correctness
    \item Very hard to use
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Goal: Speed}
  \structure{C, \cpp}
  \begin{itemize}
    \item Old (1970s, 1980s)
    \item Ubiquitous
      \begin{itemize}
        \item OSes (Linux, Windows)
        \item Games (PS, XBox, \dots)
        \item JVM
      \end{itemize}
    \item Hard to use
    \item Very fast
  \end{itemize}
  \vskip2mm
  \structure{Rust}
  \begin{itemize}
    \item Modern alternative
    \item Developed by Mozilla
    \item Hard to use
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Goal: Browser}
  \structure{JavaScript}
  \begin{itemize}
    \item Can run in browser
    \item Quick development
  \end{itemize}
  \vskip2mm
  \structure{TypeScript}
  \begin{itemize}
    \item Superset of JavaScript
    \item Compiler checks code
  \end{itemize}
  \vskip2mm
  \structure{WASM}
  \begin{itemize}
    \item Web Assembly
    \item Meant for speed
    \item Hard to use
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Goal: Software Engineering}
  \structure{Java, \csharp}
  \begin{itemize}
    \item Large scale programming
      \begin{itemize}
        \item Large code bases
        \item Large teams
      \end{itemize}
    \item Compiler checks code
    \item Fast
    \item Security
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Metrics}
  Languages differ on
  \vskip5mm
  \begin{itemize}
    \item Speed (\link{https://benchmarksgame-team.pages.debian.net/benchmarksgame/index.html}{benchmarks})
    \item Ease of use
    \item Static analysis (compiler checks for bugs)
    \item Security
    \item Supported platform
  \end{itemize}
\end{frame}
